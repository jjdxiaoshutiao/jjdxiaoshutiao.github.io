ContextMenuItemAttribute

class in UnityEngine

/

继承自： PropertyAttribute

切换到手册

描述

使用该属性可将上下文菜单添加到调用命名方法的字段。 

using UnityEngine;

public class Example : MonoBehaviour
{
    [ContextMenuItem("Reset", "ResetBiography")]
    [Multiline(8)]
    string playerBiography = "";

    void ResetBiography()
    {
        playerBiography = "";
    }
}
